% CV item command
\makeatletter
\newcommand{\cvitem}[2]{%
  \vspace{\cvitemBeforeSkip}%
  \begingroup%
  \def\cvitem@date{#1}%
  \def\cvitem@nil{\cvitem@nil}%
  \let\cvitem@firstskip\@empty%
  %
  \long\def\cvitem@split##1\\{%
    \def\cvitem@head{##1}%
    \@ifnextchar[{\cvitem@split@opt}{\cvitem@split@noopt}%
  }%
  \long\def\cvitem@split@opt[##1]##2\cvitem@nil{%
    \def\cvitem@firstskip{##1}%
    \def\cvitem@rest{##2}%
  }%
  \long\def\cvitem@split@noopt##1\cvitem@nil{%
    \def\cvitem@rest{##1}%
  }%
  %
  \cvitem@split #2\\\cvitem@nil%
  %
  \noindent%
  \begin{minipage}[t]{\dimexpr\linewidth-\cvdateWidth-0.35cm\relax}%
    \raggedright\cvitem@head%
  \end{minipage}%
  \hfill%
  \begin{minipage}[t]{\cvdateWidth}%
    \raggedleft\cvitem@date%
  \end{minipage}%
  \par%
  %
  \ifx\cvitem@rest\@empty%
    \vspace{\cvitemAfterSkip}%
  \else%
    \vspace{\cvitemBodySkip}%
    \begin{onecolentry}%
      \begingroup%
        \def\\{\@ifnextchar[{\cvitem@cr@opt}{\cvitem@cr@noopt}}%
        \def\cvitem@cr@opt[##1]{\par\vspace{##1}}%
        \def\cvitem@cr@noopt{\par}%
        \ifx\cvitem@firstskip\@empty\else\par\vspace{\cvitem@firstskip - 0.2em}\fi%
        \leavevmode\ignorespaces \cvitem@rest%
      \endgroup%
    \end{onecolentry}%
    \vspace{\cvitemAfterSkip}%
  \fi%
  \endgroup%
}
