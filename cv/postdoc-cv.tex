% cvitem variants
\makeatletter
\newcommand{\cvitemtwocol}[2]{%
  \begingroup
  \def\cvitem@date{#1}%
  \def\cvitem@nil{\cvitem@nil}%
  \let\cvitem@firstskip\@empty

  % Split at the FIRST \\ and grab [len] ONLY if it belongs to that \\.
  \long\def\cvitem@split##1\\{%
    \def\cvitem@head{##1}%
    \@ifnextchar[{\cvitem@split@opt}{\cvitem@split@noopt}%
  }%
  \long\def\cvitem@split@opt[##1]##2\cvitem@nil{%
    \def\cvitem@firstskip{##1}%
    \def\cvitem@rest{##2}%
  }%
  \long\def\cvitem@split@noopt##1\cvitem@nil{%
    \def\cvitem@rest{##1}%
  }%

  % Sentinel so split always succeeds
  \cvitem@split #2\\\cvitem@nil

  % Two-column header row
  \begin{twocolentry}{\cvitem@date}%
    \cvitem@head
  \end{twocolentry}%

  % ---- ONLY add remainder block if there is remainder ----
  \ifx\cvitem@rest\@empty
    \vspace{\cvitemskip}%
  \else
    \vspace{0.15em}%
    \begin{onecolentry}%
      \begingroup
        \def\\{%
          \@ifnextchar[{\cvitem@cr@opt}{\cvitem@cr@noopt}%
        }%
        \def\cvitem@cr@opt[##1]{\par\vspace{##1}}%
        \def\cvitem@cr@noopt{\par}%
        \ifx\cvitem@firstskip\@empty\else\par\vspace{\cvitem@firstskip - 0.2em}\fi
        \leavevmode\ignorespaces \cvitem@rest
      \endgroup
    \end{onecolentry}%
    \vspace{\cvitemskip}%
  \fi
  % -------------------------------------------------------

  \endgroup
}
\makeatother


\makeatletter
\newcommand{\cvitembullet}[2]{%
  \begingroup
  \def\cvitem@date{#1}%
  \def\cvitem@nil{\cvitem@nil}%

  \long\def\cvitem@split##1\\##2\cvitem@nil{%
    \def\cvitem@head{##1}%
    \def\cvitem@rest{##2}%
  }%
  \cvitem@split #2\\\cvitem@nil

  % Two-column header with bullet
  \onecolentry
  \setcolumnwidth{\fill, 3.5cm}%
  \begin{paracol}{2}%
    \begin{list}{\textbullet}{%
      \setlength{\leftmargin}{1.2em}%
      \setlength{\itemindent}{0pt}%
      \setlength{\topsep}{0pt}%
      \setlength{\parsep}{0pt}%
      \setlength{\itemsep}{0pt}%
      \setlength{\partopsep}{0pt}%
    }%
    \item \ignorespaces \cvitem@head
    \end{list}%
    \switchcolumn \raggedleft \cvitem@date
  \end{paracol}%
  \endonecolentry

  % Full-width remainder if present
  \ifx\cvitem@rest\@empty\else
    \vspace{0.15em}%
    \begin{adjustwidth}{1.2em}{0pt}%
      \ignorespaces \cvitem@rest
    \end{adjustwidth}%
  \fi
  \vspace{\cvitemskip}%
  \endgroup
}
\makeatother


\makeatletter
\newcommand{\cvitembulletsimple}[2]{%
  \begin{onecolentry}%
    \begingroup
      % make \\ safe; \\[len] -> spacing
      \def\\{%
        \@ifnextchar[{\cvitem@cr@opt}{\cvitem@cr@noopt}%
      }%
      \def\cvitem@cr@opt[##1]{\par\vspace{##1}}%
      \def\cvitem@cr@noopt{\par}%

      \begin{filledbullets}%
        \item \ignorespaces #2\unskip\textnormal{ (#1).}%
      \end{filledbullets}%
    \endgroup
  \end{onecolentry}%
  \vspace{\cvitemskip}%
}
\makeatother
